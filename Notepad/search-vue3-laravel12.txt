----- docker-entrypoint.sh -----
#!/bin/sh

# Cek apakah node_modules sudah ada
if [ ! -d "node_modules" ]; then
  echo "node_modules tidak ditemukan, menjalankan npm install..."
  npm install
else
  echo "node_modules sudah ada, melewati npm install"
fi

# Jalankan dev server
exec npm run dev
----- docker-entrypoint.sh -----

----- docker-compose.dev.yml -----
# Vue 3 — Development (Vite dev server, hot reload). Tanpa Nginx.
# Jalankan: docker compose -f docker-compose.dev.yml up
name: emm_sandbox_app_front

services:
  app:
    build:
      context: .
      dockerfile: Dockerfile.dev
    container_name: emm_sandbox_app_front_app
    working_dir: /app
    volumes:
      - .:/app
    command: sh docker-entrypoint.sh
    ports:
      - "8001:8001"
